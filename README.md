  match_count INT DEFAULT 3,

  id INT,
  title TEXT,
  blueprint JSONB,
  tags TEXT[],
  quality_score INT,

    kb.id,
    kb.title,
    kb.blueprint,

      WHERE tag ILIKE '%' || query_keyword || '%'

END;
$$;
